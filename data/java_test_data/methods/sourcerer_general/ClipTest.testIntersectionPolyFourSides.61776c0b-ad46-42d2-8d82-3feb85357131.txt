public void testIntersectionPolyFourSides() {
    Poly p1 = new PolyDefault();
    p1.add(1.0, 0.5);
    p1.add(5.0, 0.5);
    p1.add(5.0, 6.0);
    p1.add(3.0, 4.0);
    p1.add(1.0, 6.0);
    Poly p2 = new PolyDefault();
    p2.add(0.0, 2.0);
    p2.add(2.0, 0.0);
    p2.add(3.0, 1.0);
    p2.add(4.0, 0.0);
    p2.add(6.0, 2.0);
    p2.add(4.0, 4.0);
    p2.add(3.0, 3.0);
    p2.add(2.0, 4.0);
    Poly exp = new PolyDefault();
    exp.add(1.0, 1.0);
    exp.add(1.5, 0.5);
    exp.add(2.5, 0.5);
    exp.add(3.0, 1.0);
    exp.add(3.5, 0.5);
    exp.add(4.5, 0.5);
    exp.add(5.0, 1.0);
    exp.add(5.0, 3.0);
    exp.add(4.0, 4.0);
    exp.add(3.0, 3.0);
    exp.add(2.0, 4.0);
    exp.add(1.0, 3.0);
    Poly result = Clip.intersection(p1, p2);
    assertEquals(exp, result);
}
