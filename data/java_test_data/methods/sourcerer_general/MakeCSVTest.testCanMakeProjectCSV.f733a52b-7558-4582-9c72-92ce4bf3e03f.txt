public void testCanMakeProjectCSV() throws SQLException, IOException, NotFoundException, ClassNotFoundException {
    global.ConnectionManager.getInstance().setConnection(new MySqlConnector(new MySqlConfiguration()));
    mcsv = new MakeCSV();
    projects = new ArrayList<Project>();
    people = new ArrayList<Person>();
    milestones = new ArrayList<Milestone>();
    deliverables = new ArrayList<Deliverable>();
    publications = new ArrayList<Publication>();
    events = new ArrayList<Event>();
    d = new Date();
    p = new Person("Ben", "N", "10029", "514-555-4499", "Developper", Sex.Male, "Test Person");
    pe = new Person("John", "Doe", "12320", "514-554-2499", "Manager", Sex.Male, "Test Person 2");
    p.setCreatedAt(d);
    pe.setCreatedAt(d);
    people.add(p);
    people.add(pe);
    mil = new Milestone(3, d, p, "Test Milestone", 1, d, d, Milestone.Status.Complete, deliverables, "This is a test Milestone");
    milestones.add(mil);
    del = new Deliverable(3, p1, d, p, d, p, "Test Deliverable", 1, p, Deliverable.Status.Complete, "This is a test Deliverable");
    deliverables.add(del);
    pub = new Publication(2, d, p, "Test Publication", "Montreal", d, people, "This is a test publication");
    pub.setCreatedAt("Mon Aug 10 16:34:59 EDT 2009");
    pub.setAuthors(people);
    publications.add(pub);
    p1 = new Project(3, "Test Project", "A Test Project", p, 1, 1, 1, 1, 100, d, d, people, milestones, publications, deliverables, Status.Complete, "This is a test project");
    p2 = new Project(4, "Another Project", "A second Project", p, 1, 1, 1, 1, 75, d, d, people, milestones, publications, deliverables, Status.In_Progress, "This is also a test project");
    projects.add(p1);
    projects.add(p2);
    e = new Event(2, "Test Event", d, "This is a test event", people, Event.Status.Complete, "Montreal", p);
    e.setProject(p2);
    events.add(e);
    deliverables.clear();
    del = new Deliverable(3, p1, d, p, d, p, "Test Deliverable", 1, p, Deliverable.Status.Complete, "This is a test Deliverable");
    deliverables.add(del);
    mcsv.makeProjectReport(projects);
    mcsv.makePersonReport(people);
    mcsv.makePublicationReport(publications);
    mcsv.makeMilestoneReport(milestones);
    mcsv.makeDeliverableReport(deliverables);
    mcsv.makeEventReport(events);
    assertTrue(true);
}
