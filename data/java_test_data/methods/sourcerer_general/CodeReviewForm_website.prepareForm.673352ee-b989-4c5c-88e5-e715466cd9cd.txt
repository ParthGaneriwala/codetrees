private void prepareForm(Composite parent) {
    Group codeReview = new Group(parent, SWT.NONE);
    codeReview.setText("Create new topic");
    GridLayout gl = new GridLayout();
    gl.numColumns = 3;
    codeReview.setLayout(gl);
    setSpan(codeReview, 1, 1, GridData.FILL_HORIZONTAL);
    setSpan(createLabel(codeReview, "Topic title:"), 3, 1, GridData.BEGINNING | GridData.FILL_HORIZONTAL);
    titleText = new Text(codeReview, SWT.SINGLE | SWT.BORDER);
    setSpan(titleText, 3, 1, GridData.FILL_HORIZONTAL);
    setSpan(createLabel(codeReview, "Topic description:"), 3, 1, GridData.BEGINNING | GridData.FILL_HORIZONTAL);
    descriptionText = new Text(codeReview, SWT.MULTI | SWT.BORDER | SWT.V_SCROLL | SWT.H_SCROLL);
    setSpan(descriptionText, 3, 20, GridData.FILL_BOTH);
    createLabel(codeReview, "Topic text upload:");
    fileText = new Text(codeReview, SWT.SINGLE | SWT.BORDER | SWT.READ_ONLY);
    setSpan(fileText, 1, 1, GridData.BEGINNING | GridData.FILL_HORIZONTAL);
    Button chooseFile = new Button(codeReview, SWT.PUSH);
    chooseFile.setText("Choose file");
    createLabel(codeReview, "Repository:");
    repository = new Combo(codeReview, SWT.DROP_DOWN | SWT.READ_ONLY);
    setSpan(repository, 2, 1, GridData.FILL);
    createLabel(codeReview, "Project:");
    project = new Combo(codeReview, SWT.DROP_DOWN | SWT.READ_ONLY);
    setSpan(project, 2, 1, GridData.FILL);
    createLabel(codeReview, "Bug IDs:");
    bugId = new Text(codeReview, SWT.SINGLE | SWT.BORDER);
    setSpan(bugId, 2, 1, GridData.FILL_HORIZONTAL);
    createLabel(codeReview, "Your email address: ");
    yourEmail = new Text(codeReview, SWT.SINGLE | SWT.BORDER);
    setSpan(yourEmail, 2, 1, GridData.FILL_HORIZONTAL);
    createLabel(codeReview, "Reviewers:");
    reviewers = new Text(codeReview, SWT.SINGLE | SWT.BORDER);
    setSpan(reviewers, 2, 1, GridData.FILL_HORIZONTAL);
    createLabel(codeReview, "Cc: ");
    cc = new Text(codeReview, SWT.SINGLE | SWT.BORDER);
    setSpan(cc, 2, 1, GridData.FILL_HORIZONTAL);
    Button createTopic = new Button(codeReview, SWT.PUSH);
    createTopic.setText("Create Topic");
}
