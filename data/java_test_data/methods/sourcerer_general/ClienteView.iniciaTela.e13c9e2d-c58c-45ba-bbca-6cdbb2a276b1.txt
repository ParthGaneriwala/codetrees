public void iniciaTela() throws NumberFormatException, IOException {
    Sistema s = Sistema.getUnicaInstancia();
    cliente = ClienteControlador.get(s.getAtualId());
    ouvinte = new acao();
    tela = new JFrame();
    tela.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
    tela.setResizable(false);
    tela.setBounds(500, 300, 600, 300);
    tela.setTitle("Cliente");
    cabecalho = new JLabel("Bem Vindo Sr(a). " + cliente.getNome());
    botaoSacar = new JButton("Sacar");
    botaoSacar.addActionListener(ouvinte);
    botaoDepositar = new JButton("Depositar");
    botaoDepositar.addActionListener(ouvinte);
    botaoTransferir = new JButton("Transferir");
    botaoTransferir.addActionListener(ouvinte);
    botaoVerSaldo = new JButton("Ver Saldo");
    botaoVerSaldo.addActionListener(ouvinte);
    botaoExtrato = new JButton("Ver Extrato");
    botaoExtrato.addActionListener(ouvinte);
    botaoCriarConta = new JButton("Criar Contas");
    botaoCriarConta.addActionListener(ouvinte);
    botaoTrocarSenha = new JButton("Trocar Senha");
    botaoTrocarSenha.addActionListener(ouvinte);
    deletarConta = new JButton("Deletar Conta");
    deletarConta.addActionListener(ouvinte);
    botaoLogout = new JButton("Logout");
    botaoLogout.addActionListener(ouvinte);
    painel = new JPanel();
    painel.setLayout(new GridLayout(3, 3));
    painel.add(botaoSacar);
    painel.add(botaoDepositar);
    painel.add(botaoTransferir);
    painel.add(botaoVerSaldo);
    painel.add(botaoExtrato);
    painel.add(botaoCriarConta);
    painel.add(deletarConta);
    painel.add(botaoTrocarSenha);
    painel.add(botaoLogout);
    tela.add(cabecalho, BorderLayout.NORTH);
    tela.add(painel, BorderLayout.CENTER);
    tela.setVisible(true);
}
