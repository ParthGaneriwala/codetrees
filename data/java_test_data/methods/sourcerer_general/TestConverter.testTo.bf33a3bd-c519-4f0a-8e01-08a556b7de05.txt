public static void testTo(Converter c) {
    System.out.println("TO");
    Point2f[] p = new Point2f[10];
    boolean ehto;
    int i = 0, x, y;
    p[i] = c.to(new Point2f(150, 100));
    ehto = (int) p[i].getX() == 0 && (int) p[i].getY() == 0;
    assert ehto : "\nTest " + i + " failed " + p[i];
    i++;
    p[i] = c.to(new Point2f(300, 100));
    x = (int) p[i].getX();
    y = (int) p[i].getY();
    ehto = x == 10 && y == 0;
    assert ehto : "\nTest " + i + " failed " + p[i];
    i++;
    p[i] = c.to(new Point2f(0, 100));
    x = (int) p[i].getX();
    y = (int) p[i].getY();
    ehto = x == -10 && y == 0;
    assert ehto : "\nTest " + i + " failed " + p[i];
    i++;
    p[i] = c.to(new Point2f(0, 0));
    x = (int) p[i].getX();
    y = (int) p[i].getY();
    ehto = x == -10 && y == 10;
    assert ehto : "\nTest " + i + " failed " + p[i];
    i++;
    p[i] = c.to(new Point2f(150, 200));
    x = (int) p[i].getX();
    y = (int) p[i].getY();
    ehto = x == 0 && y == -10;
    assert ehto : "\nTest " + i + " failed " + p[i];
    i++;
    p[i] = c.to(new Point2f(225, 50));
    x = (int) p[i].getX();
    y = (int) p[i].getY();
    ehto = x == 5 && y == 5;
    assert ehto : "\nTest " + i + " failed " + p[i];
    i++;
    p[i] = c.to(new Point2f(225, 150));
    x = (int) p[i].getX();
    y = (int) p[i].getY();
    ehto = x == 5 && y == -5;
    assert ehto : "\nTest " + i + " failed " + p[i];
    i++;
    p[i] = c.to(new Point2f(300, 200));
    x = (int) p[i].getX();
    y = (int) p[i].getY();
    ehto = x == 10 && y == -10;
    assert ehto : "\nTest " + i + " failed " + p[i];
    i++;
    p[i] = c.to(new Point2f(0, 0));
    x = (int) p[i].getX();
    y = (int) p[i].getY();
    ehto = x == -10 && y == 10;
    assert ehto : "\nTest " + i + " failed " + p[i];
    i++;
    System.out.println("No errors found.");
}
