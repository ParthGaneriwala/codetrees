private void composePanel() {
    setupActions();
    JPanel mainPanel = new JPanel(new GridBagLayout());
    JPanel containerPanel = new JPanel(new GridBagLayout());
    mCreateWaypointsBtn = new JButton(mWaypointsAxn);
    mRmvWPBtn = new JButton(mRmvWPAxn);
    mRmvAllWPBtn = new JButton(mRmvAllWPAxn);
    mSetMeetingPointBtn = new JButton(mMeetPntAxn);
    mRmvMPBtn = new JButton(mRmvMPAxn);
    mRmvAllMPBtn = new JButton(mRmvAllMPAxn);
    mInitGPSBtn = new JButton(mInitGPSAxn);
    mainPanel.setBorder(BorderFactory.createTitledBorder(BorderFactory.createEtchedBorder(EtchedBorder.LOWERED), "GPS Interaction"));
    GridBagConstraints settings = new GridBagConstraints();
    settings.gridx = settings.gridy = 0;
    settings.weightx = 1.0;
    settings.weighty = 0;
    settings.insets = new Insets(0, 0, 0, 0);
    settings.anchor = GridBagConstraints.NORTHWEST;
    settings.fill = GridBagConstraints.HORIZONTAL;
    mainPanel.add(mInitGPSBtn, settings);
    settings.gridy++;
    mainPanel.add(mCreateWaypointsBtn, settings);
    settings.gridy++;
    mainPanel.add(mRmvWPBtn, settings);
    settings.gridy++;
    mainPanel.add(mRmvAllWPBtn, settings);
    settings.gridy++;
    mainPanel.add(mSetMeetingPointBtn, settings);
    settings.gridy++;
    mainPanel.add(mRmvMPBtn, settings);
    settings.gridy++;
    settings.insets = new Insets(0, 0, 20, 0);
    mainPanel.add(mRmvAllMPBtn, settings);
    settings.gridx = settings.gridy = 0;
    containerPanel.add(mainPanel, settings);
    if (mAuthHandler.checkTeacherAuthentication())
        composeAdmin(containerPanel, settings);
    setLayout(new GridBagLayout());
    settings.gridx = settings.gridy = 0;
    settings.fill = GridBagConstraints.NONE;
    settings.weighty = 1.0;
    add(containerPanel, settings);
}
