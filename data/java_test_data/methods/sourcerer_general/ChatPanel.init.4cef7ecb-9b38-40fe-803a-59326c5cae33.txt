public void init() {
    setLayout(new GridLayout(2, 1));
    messageHistory = new JTextArea();
    messageHistory.setLineWrap(true);
    messageHistory.setText("This is your message history pane, here you will be able to view messages that have been sent to you \n");
    messageHistory.setRows(5);
    messageHistory.setMaximumSize(messageHistory.getSize());
    JScrollPane messageScrollPane = new JScrollPane(messageHistory);
    text = new JTextArea("Enter your message here to begin discussing the map...");
    text.setLineWrap(true);
    text.setRows(2);
    JScrollPane inputScrollPane = new JScrollPane(text);
    JButton submit = new JButton("Send");
    JLabel submitLabel = new JLabel("Message: ");
    JPanel input = new JPanel();
    text.addKeyListener(new NewMessageKeyboardListener());
    submit.addMouseListener(new NewMessageMouseListener());
    input.setLayout(new GridBagLayout());
    GridBagConstraints c = new GridBagConstraints();
    c.gridx = 0;
    c.gridy = 0;
    c.weightx = 0.0;
    c.anchor = GridBagConstraints.LINE_START;
    input.add(submitLabel, c);
    c.gridx = 1;
    c.weightx = 1.0;
    c.fill = GridBagConstraints.HORIZONTAL;
    input.add(inputScrollPane, c);
    c.gridx = 2;
    c.weightx = 0.0;
    c.anchor = GridBagConstraints.LINE_END;
    input.add(submit, c);
    add(messageScrollPane);
    add(input);
}
