@Test
public void testRSRCRawData() {
    RSRC rsrc = clel.getRSRC();
    ArrayList<Directory> dir = rsrc.getDirectories();
    Directory d = dir.get(0);
    ArrayList<Entry> entries = d.getEntries();
    Entry e = entries.get(0);
    assertEquals("RSRC - Characteristics", d.getCharacteristics(), 0x0);
    assertEquals("RSRC - TimeDateStamp", d.getTimeDateStamp(), 0x0);
    assertEquals("RSRC - MajorVersion", d.getMajorVersion(), (short) 0x0);
    assertEquals("RSRC - MinorVersion", d.getMinorVersion(), (short) 0x0);
    assertEquals("RSRC - NumberOfNameEntries", d.getNumberOfNameEntries(), (short) 0x0);
    assertEquals("RSRC - NumberOfIDEntries", d.getNumberOfIDEntries(), (short) 0x1);
    assertEquals("RSRC - Entry ID", e.getID(), 0x10);
    assertEquals("RSRC - Entry Offset", e.getOffset(), 0x80000018);
    d = dir.get(1);
    entries = d.getEntries();
    e = entries.get(0);
    assertEquals("RSRC - Characteristics", d.getCharacteristics(), 0x0);
    assertEquals("RSRC - TimeDateStamp", d.getTimeDateStamp(), 0x0);
    assertEquals("RSRC - MajorVersion", d.getMajorVersion(), (short) 0x0);
    assertEquals("RSRC - MinorVersion", d.getMinorVersion(), (short) 0x0);
    assertEquals("RSRC - NumberOfNameEntries", d.getNumberOfNameEntries(), (short) 0x0);
    assertEquals("RSRC - NumberOfIDEntries", d.getNumberOfIDEntries(), (short) 0x1);
    assertEquals("RSRC - Entry ID", e.getID(), 0x1);
    assertEquals("RSRC - Entry Offset", e.getOffset(), 0x80000030);
    d = dir.get(2);
    entries = d.getEntries();
    e = entries.get(0);
    assertEquals("RSRC - Characteristics", d.getCharacteristics(), 0x0);
    assertEquals("RSRC - TimeDateStamp", d.getTimeDateStamp(), 0x0);
    assertEquals("RSRC - MajorVersion", d.getMajorVersion(), (short) 0x0);
    assertEquals("RSRC - MinorVersion", d.getMinorVersion(), (short) 0x0);
    assertEquals("RSRC - NumberOfNameEntries", d.getNumberOfNameEntries(), (short) 0x0);
    assertEquals("RSRC - NumberOfIDEntries", d.getNumberOfIDEntries(), (short) 0x1);
    assertEquals("RSRC - Entry ID", e.getID(), 0x0);
    assertEquals("RSRC - Entry Offset", e.getOffset(), 0x48);
    ResourceEntry re = e.getData();
    assertEquals("RSRC - ResourceEntry DataRVA", re.getDataRVA(), 0x4058);
    assertEquals("RSRC - ResourceEntry Size", re.getSize(), 0x27c);
    assertEquals("RSRC - ResourceEntry CodePage", re.getCodePage(), 0x0);
    assertEquals("RSRC - ResourceEntry Reserved", re.getReserved(), 0x0);
    ArrayList<Byte> data = re.getData();
    assertEquals("RSRC - ResourceEntry Data", data.size(), 0x27c);
}
