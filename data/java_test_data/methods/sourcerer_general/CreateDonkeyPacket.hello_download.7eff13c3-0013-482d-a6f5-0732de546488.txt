public static donkeyPacket hello_download(donkeyInfo di) {
    donkeyPacket myDp = new donkeyPacket();
    myDp.setCommandId(convert.intToByte(0x01));
    byte[] usernameBytes = new byte[di.getUserName().length()];
    usernameBytes = di.getUserName().getBytes();
    byte[] usernameLengthBytes = new byte[2];
    int usernameLength = usernameBytes.length;
    usernameLengthBytes = convert.intToUShort(usernameLength);
    byte[] oldUseridBytes = new byte[4];
    oldUseridBytes = convert.intToDWORD((int) di.getUserID());
    byte[] inPortBytes = new byte[2];
    inPortBytes = convert.intToUShort(di.getInPort());
    byte[] clientVersionBytes = new byte[4];
    clientVersionBytes = convert.intToDWORD((int) di.getClientVersion());
    byte[] userHashBytes = di.getUserHash();
    byte[] packetBody = new byte[47 + usernameLength];
    packetBody[0] = convert.intToByte(16);
    System.arraycopy(userHashBytes, 0, packetBody, 1, 16);
    // System.arraycopy(oldUseridBytes,0,packetBody,16,4);
    packetBody[17] = convert.intToByte(1);
    packetBody[18] = convert.intToByte(0);
    packetBody[19] = convert.intToByte(0);
    packetBody[10] = convert.intToByte(0);
    System.arraycopy(inPortBytes, 0, packetBody, 21, 2);
    packetBody[23] = convert.intToByte(0x02);
    packetBody[24] = convert.intToByte(0x00);
    packetBody[25] = convert.intToByte(0x00);
    packetBody[26] = convert.intToByte(0x00);
    packetBody[27] = convert.intToByte(0x02);
    packetBody[28] = convert.intToByte(0x01);
    packetBody[29] = convert.intToByte(0x00);
    packetBody[30] = convert.intToByte(0x01);
    System.arraycopy(usernameLengthBytes, 0, packetBody, 31, 2);
    System.arraycopy(usernameBytes, 0, packetBody, 33, usernameLength);
    packetBody[33 + usernameLength] = convert.intToByte(0x03);
    packetBody[34 + usernameLength] = convert.intToByte(0x01);
    packetBody[35 + usernameLength] = convert.intToByte(0x00);
    packetBody[36 + usernameLength] = convert.intToByte(0x11);
    System.arraycopy(clientVersionBytes, 0, packetBody, 37 + usernameLength, 4);
    byte[] ip_tmp = convert.intToDWORD((int) di.getUserID());
    System.arraycopy(ip_tmp, 0, packetBody, 41 + usernameLength, 4);
    System.arraycopy(inPortBytes, 0, packetBody, 45 + usernameLength, 2);
    myDp.setBody(packetBody);
    return (myDp);
}
