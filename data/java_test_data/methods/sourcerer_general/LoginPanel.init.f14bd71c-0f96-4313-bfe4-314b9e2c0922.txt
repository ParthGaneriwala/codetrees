public void init() {
    window = new JPanel(new GridBagLayout());
    window.setBackground(Color.white);
    GridBagConstraints constraints = new GridBagConstraints();
    constraints.ipadx = 10;
    mainTitle = new JTextArea("Welcome to the CH@TROOM");
    mainTitle.setAlignmentX(CENTER_ALIGNMENT);
    mainTitle.setEditable(false);
    constraints.gridx = 0;
    constraints.gridy = 0;
    window.add(mainTitle, constraints);
    instructionLabel = new JLabel("Please enter your username:");
    instructionLabel.setAlignmentX(CENTER_ALIGNMENT);
    constraints.gridx = 0;
    constraints.gridy = 1;
    window.add(instructionLabel, constraints);
    inputField = new JTextField("<username>");
    inputField.setEnabled(false);
    inputField.setColumns(10);
    inputField.addActionListener(this);
    constraints.gridx = 0;
    constraints.gridy = 2;
    window.add(inputField, constraints);
    connectButton = new JButton("Log in");
    connectButton.setAlignmentX(CENTER_ALIGNMENT);
    connectButton.addActionListener(this);
    constraints.gridx = 0;
    constraints.gridy = 3;
    window.add(connectButton, constraints);
    errorLabel = new JLabel("");
    errorLabel.setForeground(Color.red);
    errorLabel.setAlignmentX(CENTER_ALIGNMENT);
    constraints.gridx = 0;
    constraints.gridy = 4;
    window.add(errorLabel, constraints);
    connectedLabel = new JLabel("Not connected");
    connectedLabel.setForeground(Color.red);
    connectedLabel.setAlignmentX(CENTER_ALIGNMENT);
    constraints.gridx = 0;
    constraints.gridy = 5;
    window.add(connectedLabel, constraints);
    add(window);
}
