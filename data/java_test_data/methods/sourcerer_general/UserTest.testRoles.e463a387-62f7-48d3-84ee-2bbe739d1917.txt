public void testRoles() {
    Role emptyRole = user.getRole();
    assertFalse(emptyRole.hasRole(Roles.ADMINISTRATOR));
    assertFalse(emptyRole.hasRole(Roles.DEVELOPMENT_MANAGER));
    assertFalse(emptyRole.hasRole(Roles.DEVELOPER));
    assertFalse(emptyRole.hasRole(Roles.TEST_MANAGER));
    assertFalse(emptyRole.hasRole(Roles.TESTER));
    user.setRole(Roles.ADMINISTRATOR);
    assertTrue(user.hasRole(Roles.ADMINISTRATOR));
    user.setRole(Roles.TEST_MANAGER);
    assertTrue(user.hasRole(Roles.TEST_MANAGER));
    user.setRole(Roles.DEVELOPMENT_MANAGER);
    assertTrue(user.hasRole(Roles.DEVELOPMENT_MANAGER));
    user.setRole(Roles.TESTER);
    assertTrue(user.hasRole(Roles.TESTER));
    user.setRole(Roles.DEVELOPER);
    assertTrue(user.hasRole(Roles.DEVELOPER));
    Role fullRole = user.getRole();
    assertFalse(emptyRole.hasRole(Roles.ADMINISTRATOR));
    assertFalse(emptyRole.hasRole(Roles.DEVELOPMENT_MANAGER));
    assertFalse(emptyRole.hasRole(Roles.DEVELOPER));
    assertFalse(emptyRole.hasRole(Roles.TEST_MANAGER));
    assertFalse(emptyRole.hasRole(Roles.TESTER));
    assertTrue(fullRole.hasRole(Roles.ADMINISTRATOR));
    assertTrue(fullRole.hasRole(Roles.DEVELOPMENT_MANAGER));
    assertTrue(fullRole.hasRole(Roles.DEVELOPER));
    assertTrue(fullRole.hasRole(Roles.TEST_MANAGER));
    assertTrue(fullRole.hasRole(Roles.TESTER));
    user.clearRole(Roles.ADMINISTRATOR);
    assertFalse(user.hasRole(Roles.ADMINISTRATOR));
    user.clearRole(Roles.TEST_MANAGER);
    assertFalse(user.hasRole(Roles.TEST_MANAGER));
    user.clearRole(Roles.DEVELOPMENT_MANAGER);
    assertFalse(user.hasRole(Roles.DEVELOPMENT_MANAGER));
    user.clearRole(Roles.TESTER);
    assertFalse(user.hasRole(Roles.TESTER));
    user.clearRole(Roles.DEVELOPER);
    assertFalse(user.hasRole(Roles.DEVELOPER));
}
