private Track createRandomGenesTrack(int inXAxisStartValue, int inXAxisEndValue) {
    Track track = new Track("Random Genes");
    track.setURL("http://google.com");
    track.setXAxisStartValue(inXAxisStartValue);
    track.setXAxisEndValue(inXAxisEndValue);
    track.setShowId(true);
    Gene2D gene = new Gene2D();
    gene.addExon(new Exon(10, 50, Strand.FORWARD));
    gene.addExon(new Exon(70, 102, Strand.FORWARD));
    gene.addExon(new Exon(130, 190, Strand.FORWARD));
    gene.setId("CRC120");
    gene.setDescription("The random gibberish gene");
    track.addGene(gene);
    gene = new Gene2D();
    gene.setColor(HTMLColor.RED);
    gene.addExon(new Exon(10, 50, Strand.FORWARD));
    gene.addExon(new Exon(70, 90, Strand.FORWARD));
    gene.addExon(new Exon(130, 195, Strand.FORWARD));
    gene.setId("Important");
    gene.setDescription("The random gibberish accessory gene");
    gene.addURL(new Link("URL1", "Link 1"));
    gene.addURL(new Link("URL2", "Link 2"));
    gene.addURL(new Link());
    gene.addURL(new Link("URL3", "Link 3"));
    gene.addURL(new Link("URL4", "Link 4"));
    gene.addURL(new Link(null, "More Options"));
    gene.addURL(new Link("URL5", "Option 1"));
    gene.addURL(new Link("URL6", "Option 2"));
    track.addGene(gene);
    gene = new Gene2D();
    gene.addExon(new Exon(220, 221, Strand.FORWARD));
    gene.setId("CRC123");
    gene.setDescription("The CRC123 random gibberish accessory gene 2a");
    track.addGene(gene);
    gene = new Gene2D();
    gene.addExon(new Exon(320, 325, Strand.FORWARD));
    gene.setId("CRC124");
    gene.setDescription("The random gibberish accessory gene 2b");
    track.addGene(gene);
    gene = new Gene2D();
    gene.addExon(new Exon(329, 335, Strand.FORWARD));
    gene.setId("CRC125");
    gene.setDescription("The random gibberish accessory gene 2c");
    track.addGene(gene);
    // A gene straddling the right side of the display
    gene = new Gene2D();
    gene.addExon(new Exon(inXAxisEndValue - 10, inXAxisEndValue + 10, Strand.FORWARD));
    gene.setId("CRC125");
    gene.setDescription("The random gibberish accessory gene 2d");
    track.addGene(gene);
    // A gene past the right side of the display
    gene = new Gene2D();
    gene.addExon(new Exon(inXAxisEndValue + 10, inXAxisEndValue + 20, Strand.FORWARD));
    gene.setId("CRC008");
    gene.setDescription("The gene too far to the right");
    track.addGene(gene);
    gene = new Gene2D();
    gene.addExon(new Exon(235, 220, Strand.REVERSE));
    gene.addExon(new Exon(210, 200, Strand.REVERSE));
    gene.setId("CRC1266");
    gene.setDescription("The random gibberish accessory gene 2w");
    track.addGene(gene);
    gene = new Gene2D();
    gene.addExon(new Exon(335, 320, Strand.REVERSE));
    gene.addExon(new Exon(310, 300, Strand.REVERSE));
    gene.setId("CRC126");
    gene.setDescription("The random gibberish accessory gene 2e");
    track.addGene(gene);
    gene = new Gene2D();
    gene.addExon(new Exon(320, 335, Strand.REVERSE));
    gene.addExon(new Exon(300, 310, Strand.REVERSE));
    gene.setId("CRC126");
    gene.setDescription("The random gibberish accessory gene 2f");
    track.addGene(gene);
    gene = new Gene2D();
    gene.addExon(new Exon(320, 335, Strand.FORWARD).setColor(HTMLColor.GREEN));
    gene.addExon(new Exon(300, 310, Strand.FORWARD));
    gene.setId("Mr._Green_Genes");
    gene.setDescription("A gene with a green exon");
    track.addGene(gene);
    gene = new Gene2D();
    gene.setStartLocation(700);
    gene.setEndLocation(800);
    gene.setId("Exon-less7");
    gene.setDescription("A gene w/o exons");
    track.addGene(gene);
    gene = new Gene2D();
    gene.setHeightInPixels(30);
    gene.addExon(new Exon(420, 445, Strand.FORWARD));
    gene.addExon(new Exon(400, 410, Strand.FORWARD));
    gene.setId("TallGene");
    gene.setDescription("A gene for tallness");
    track.addGene(gene);
    return track;
}
