    assertFalse(user.checkPassword("bunk"));
    assertFalse(user.isNullUser());

    assertTrue("removeDefectCreator() returned false!", user.removeDefectCreator());

    assertTrue("removeDefectTester() returned false!", user.removeDefectTester());
    assertEquals("removeDefectTester() failed!", user.getNumDefectsTester(), 0);
    assertTrue("removeDefectTester() should fail!", !user.removeDefectTester());
    user.addDefectDeveloper();

    assertEquals("removeDefectDeveloper() failed!", user.getNumDefectsDeveloper(), 0);
    assertTrue("removeDefectDeveloper() should fail!", !user.removeDefectDeveloper());
    assertEquals(user.getNumTestGroupAssignments(), 0);
