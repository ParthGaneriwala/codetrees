    province = new Province();
    province.setName("Tucuman");
    province.setBusinessPK("TUC");
    City city = new City();
    city.setName("San Miguel de Tucuman");
    city.setBusinessPK("TUC");
    Region region = new Region();
    region.setName("Barrio los Decanos");
    region.setBusinessPK("BDE");
    city.addRegion(region);
    region = new Region();
    region.setName("Barrio Parque SUR");
    region.setBusinessPK("BPS");
    city.addRegion(region);
    province.addCity(city);
    provinceDAO.saveProvince(province);
    assertNotNull("primary key assigned", province.getId());
    log.info(province);
