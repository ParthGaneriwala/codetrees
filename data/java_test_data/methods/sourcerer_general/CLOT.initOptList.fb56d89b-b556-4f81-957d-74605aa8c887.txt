private void initOptList() {
    contentPane = new JPanel();
    contentPane.setLayout(new BoxLayout(contentPane, BoxLayout.PAGE_AXIS));
    selection = new JPanel();
    selection.setLayout(new BoxLayout(selection, BoxLayout.LINE_AXIS));
    sel_panel = new JPanel();
    sel_panel.setLayout(new BoxLayout(sel_panel, BoxLayout.PAGE_AXIS));
    sel = new JList(new DefaultListModel());
    DefaultListModel dlm = (DefaultListModel) sel.getModel();
    int i;
    for (i = 0; i < opts.length; i++) dlm.addElement(opts[i]);
    sel_pane = new JScrollPane(sel);
    sel_pane.setBorder(BorderFactory.createLineBorder(Color.black));
    sel_pane.setPreferredSize(new Dimension(190, 210));
    sel_panel.add(new JLabel("Available Optimization(s)"));
    sel_panel.add(sel_pane);
    selection.add(sel_panel);
    initAddButtons();
    order_panel = new JPanel();
    order_panel.setLayout(new BoxLayout(order_panel, BoxLayout.PAGE_AXIS));
    order = new JList(new DefaultListModel());
    order_pane = new JScrollPane(order);
    order_pane.setBorder(BorderFactory.createLineBorder(Color.black));
    order_panel.add(new JLabel("Optimization(s) To Run"));
    order_panel.add(order_pane);
    selection.add(order_panel);
    initMoveButtons();
    initRunOptimizationsButton();
    contentPane.add(selection);
    contentPane.add(buttons);
    window.add(contentPane);
}
