public void testNullUserRoles() {
    assertTrue(nullUser.isNullUser());
    assertFalse(nullUser.hasRole(Roles.ADMINISTRATOR));
    assertFalse(nullUser.hasRole(Roles.DEVELOPMENT_MANAGER));
    assertFalse(nullUser.hasRole(Roles.DEVELOPER));
    assertFalse(nullUser.hasRole(Roles.TEST_MANAGER));
    assertFalse(nullUser.hasRole(Roles.TESTER));
    nullUser.setRole(Roles.ADMINISTRATOR);
    assertFalse(nullUser.hasRole(Roles.ADMINISTRATOR));
    nullUser.setRole(Roles.DEVELOPMENT_MANAGER);
    assertFalse(nullUser.hasRole(Roles.DEVELOPMENT_MANAGER));
    nullUser.setRole(Roles.DEVELOPER);
    assertFalse(nullUser.hasRole(Roles.DEVELOPER));
    nullUser.setRole(Roles.TEST_MANAGER);
    assertFalse(nullUser.hasRole(Roles.TEST_MANAGER));
    nullUser.setRole(Roles.TESTER);
    assertFalse(nullUser.hasRole(Roles.TESTER));
    nullUser.clearRole(Roles.ADMINISTRATOR);
    assertFalse(nullUser.hasRole(Roles.ADMINISTRATOR));
    nullUser.clearRole(Roles.DEVELOPMENT_MANAGER);
    assertFalse(nullUser.hasRole(Roles.DEVELOPMENT_MANAGER));
    nullUser.clearRole(Roles.DEVELOPER);
    assertFalse(nullUser.hasRole(Roles.DEVELOPER));
    nullUser.clearRole(Roles.TEST_MANAGER);
    assertFalse(nullUser.hasRole(Roles.TEST_MANAGER));
    nullUser.clearRole(Roles.TESTER);
    assertFalse(nullUser.hasRole(Roles.TESTER));
}
