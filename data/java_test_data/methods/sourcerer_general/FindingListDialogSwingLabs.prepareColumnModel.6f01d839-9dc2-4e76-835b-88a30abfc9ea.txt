private TableColumnModel prepareColumnModel() {
    TableColumnModel columnModel = new DefaultTableColumnModel();
    TableColumn packageColumn = new TableColumn(0);
    packageColumn.setHeaderValue("Package");
    packageColumn.setPreferredWidth(30);
    TableColumn fileColumn = new TableColumn(1);
    fileColumn.setHeaderValue("File");
    fileColumn.setPreferredWidth(30);
    TableColumn typeColumn = new TableColumn(2);
    typeColumn.setHeaderValue("Type");
    typeColumn.setPreferredWidth(30);
    TableColumn lineNumberColumn = new TableColumn(3);
    lineNumberColumn.setHeaderValue("Line (check)");
    lineNumberColumn.setCellRenderer(new CenterRenderer());
    lineNumberColumn.setPreferredWidth(10);
    TableColumn metricValueColumn = new TableColumn(4);
    metricValueColumn.setHeaderValue("MetricValue");
    metricValueColumn.setCellRenderer(new CenterRenderer());
    metricValueColumn.setPreferredWidth(10);
    TableColumn messageColumn = new TableColumn(5);
    messageColumn.setHeaderValue("Description");
    messageColumn.setPreferredWidth(50);
    columnModel.addColumn(packageColumn);
    columnModel.addColumn(fileColumn);
    columnModel.addColumn(typeColumn);
    columnModel.addColumn(lineNumberColumn);
    columnModel.addColumn(metricValueColumn);
    columnModel.addColumn(messageColumn);
    return columnModel;
}
