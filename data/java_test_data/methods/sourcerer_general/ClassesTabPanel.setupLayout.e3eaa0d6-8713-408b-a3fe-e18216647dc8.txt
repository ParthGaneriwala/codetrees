private void setupLayout() {
    GridBagConstraints settings = new GridBagConstraints(0, 0, 2, 4, 1.0, 0.5, GridBagConstraints.CENTER, GridBagConstraints.BOTH, new Insets(6, 6, 6, 6), 0, 0);
    viewClass.setEnabled(false);
    classesButtonsPanel.add(viewClass, settings);
    settings.gridy = 1;
    classesButtonsPanel.add(newClass, settings);
    settings.gridy = 2;
    deleteClass.setEnabled(false);
    classesButtonsPanel.add(deleteClass, settings);
    settings.gridy = 3;
    // classesButtonsPanel.add(editClass, settings);
    // settings.gridy = 4;
    addClassStudents.setEnabled(false);
    classesButtonsPanel.add(addClassStudents, settings);
    viewGroup.setEnabled(false);
    groupsButtonsPanel.add(viewGroup, settings);
    settings.gridy = 1;
    groupsButtonsPanel.add(newGroup, settings);
    settings.gridy = 2;
    deleteGroup.setEnabled(false);
    groupsButtonsPanel.add(deleteGroup, settings);
    settings.gridy = 3;
    // groupsButtonsPanel.add(editGroups, settings);
    // settings.gridy = 4;
    addGroupStudents.setEnabled(false);
    groupsButtonsPanel.add(addGroupStudents, settings);
    this.add(classesPanel, new GridBagConstraints(0, 0, 2, 4, 1.0, 0.5, GridBagConstraints.CENTER, GridBagConstraints.BOTH, new Insets(6, 6, 6, 6), 0, 0));
    this.add(classesButtonsPanel, new GridBagConstraints(0, 1, 2, 4, 1.0, 0.5, GridBagConstraints.CENTER, GridBagConstraints.BOTH, new Insets(6, 6, 6, 6), 0, 0));
    this.add(groupsPanel, new GridBagConstraints(0, 2, 2, 4, 1.0, 0.5, GridBagConstraints.CENTER, GridBagConstraints.BOTH, new Insets(6, 6, 6, 6), 0, 0));
    this.add(groupsButtonsPanel, new GridBagConstraints(0, 3, 2, 4, 1.0, 0.5, GridBagConstraints.CENTER, GridBagConstraints.BOTH, new Insets(6, 6, 6, 6), 0, 0));
}
