public void testStalemate() {
    assertFalse(gl.checkForStalemate(gb));
    assertFalse(gl.checkForWin(gb, 0, 0));
    int col;
    gb.makeMove(gl.getTurn().generatePiece(), 0);
    assertFalse(gl.checkForWin(gb, 0, 0));
    gl.nextTurn();
    gb.makeMove(gl.getTurn().generatePiece(), 2);
    assertFalse(gl.checkForWin(gb, 2, 0));
    gl.nextTurn();
    gb.makeMove(gl.getTurn().generatePiece(), 1);
    assertFalse(gl.checkForWin(gb, 1, 0));
    gl.nextTurn();
    col = gb.makeMove(gl.getTurn().generatePiece(), 0);
    assertFalse(gl.checkForWin(gb, 0, col));
    gl.nextTurn();
    col = gb.makeMove(gl.getTurn().generatePiece(), 1);
    assertFalse(gl.checkForWin(gb, 1, col));
    gl.nextTurn();
    col = gb.makeMove(gl.getTurn().generatePiece(), 1);
    assertFalse(gl.checkForWin(gb, 1, col));
    gl.nextTurn();
    col = gb.makeMove(gl.getTurn().generatePiece(), 2);
    assertFalse(gl.checkForWin(gb, 2, col));
    gl.nextTurn();
    col = gb.makeMove(gl.getTurn().generatePiece(), 2);
    assertFalse(gl.checkForWin(gb, 2, col));
    gl.nextTurn();
    assertFalse(gl.checkForStalemate(gb));
    col = gb.makeMove(gl.getTurn().generatePiece(), 0);
    assertFalse(gl.checkForWin(gb, 0, col));
    assertTrue(gl.checkForStalemate(gb));
}
