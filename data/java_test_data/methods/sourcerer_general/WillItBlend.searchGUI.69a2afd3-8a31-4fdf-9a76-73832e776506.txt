private static JFrame searchGUI(Collection<CatalogItem> items) {
    frame.setVisible(false);
    frame = new JFrame("Hedlund International");
    JTabbedPane body = new JTabbedPane();
    CartListener cListen = new CartListener(cart);
    JPanel indirection = new JPanel();
    Checkout x = new Checkout(cart.recipient());
    search = new Search(null);
    ScrolledResultList results = new ScrolledResultList(new ResultList("Results", items));
    results.content().behavior(cListen);
    search.command(SearchListener.get(frame, catalog));
    indirection.setLayout(new BorderLayout());
    indirection.add(search, BorderLayout.NORTH);
    indirection.add(results);
    CategorySelector cs = new CategorySelector().command(SearchListener.get(frame, catalog));
    indirection.add(cs, BorderLayout.WEST);
    body.addTab("Search", indirection);
    CartList cartDisp = cartGUI();
    cListen.addView(cartDisp);
    body.addTab("Cart", cartDisp);
    body.addTab("Checkout", x);
    x.priceDisp().setAction(new Checkout.CheckoutComputer(x, cart));
    body.addChangeListener(new ChangeTab(frame));
    frame.add(body);
    SearchListener.get(frame, catalog).prepare(search, cs, results);
    frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
    frame.pack();
    frame.setVisible(true);
    return frame;
}
