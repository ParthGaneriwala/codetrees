protected void initAcroPlugin() {
    AcroPlugContainer = createContainer("", UIProvider.HIDEABLE_CONTAINER_MODE);
    AcroPlugContainer.setTitle("Acro Plugin");
    AcroPlugContainer.addPluginUIListener(this);
    AcroPlugContainer.addMenu(createPluginMenu());
    getConn().getListenerManager().addChatListener(this);
    AcroPlugContainer.getContentPane().setLayout(new FlowLayout());
    AcroPlugContainer.getContentPane().setBackground(Color.GREEN);
    AcroPan = new AcroPanel();
    AcroPan.setSize(800, 150);
    AcroPan.setBackground(Color.RED);
    AcroPlugContainer.getContentPane().add(AcroPan);
    AcroPlugContainer.getContentPane().paintAll(AcroPlugContainer.getContentPane().getGraphics());
    TimeText = new JTextField("Time:", 10);
    TimeText.setFont(new Font("Monospaced", Font.BOLD, 22));
    AcroPlugContainer.getContentPane().add(TimeText);
    AcroText = new JTextArea("Acrobot Text" + CR, 8, 64);
    AcroPlugContainer.getContentPane().add(new JScrollPane(AcroText));
    AcroTellText = new JTextField("!topten", 24);
    AcroTellText.addKeyListener(this);
    AcroTellText.setFont(new Font("Serif", Font.PLAIN, 16));
    AcroPlugContainer.getContentPane().add(AcroTellText);
    // SubButt = new JButton("Submit"); SubButt.addActionListener(this);
    // AcroPlugContainer.getContentPane().add(SubButt);
    ChanText = new JTextArea("Acro Channel (" + acroChan + ") Text" + CR, 8, 64);
    AcroPlugContainer.getContentPane().add(new JScrollPane(ChanText));
    ChanTellText = new JTextField("Hello!", 24);
    ChanTellText.addKeyListener(this);
    ChanTellText.setFont(new Font("Serif", Font.PLAIN, 16));
    AcroPlugContainer.getContentPane().add(ChanTellText);
    VoteBox = new JList(acrolist);
    // GraphicsEnvironment.getLocalGraphicsEnvironment().getAvailableFontFamilyNames());
    VoteBox.setFont(new Font("Times New Roman", Font.BOLD, 18));
    VoteBox.addMouseListener(this);
    AcroPlugContainer.getContentPane().add(new JScrollPane(VoteBox));
}
