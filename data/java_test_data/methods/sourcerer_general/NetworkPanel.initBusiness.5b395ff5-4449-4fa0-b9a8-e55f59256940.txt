public void initBusiness() {
    box = new TDataBox();
    network = new TNetwork(box);
    network.setBackground(new ColorBackground(new Color(153, 204, 255)));
    add(network);
    Node crm = new Node("crm");
    crm.setImage("/com/starchoice/itm/client/core/server.png");
    crm.setName("公司CRM系统");
    crm.setLocation(50, 500);
    box.addElement(crm);
    Node erp = new Node("tel");
    erp.setImage("/com/starchoice/itm/client/core/server.png");
    erp.setName("公司ERP系统");
    erp.setLocation(160, 500);
    box.addElement(erp);
    Node nodeA = new Node("A");
    nodeA.setImage("/com/starchoice/itm/client/core/cisco-l3switch-32.png");
    nodeA.setName("思科核心路由器");
    nodeA.setLocation(50, 300);
    box.addElement(nodeA);
    Node nodeB = new Node("B");
    nodeB.setImage("/com/starchoice/itm/client/core/huawei-common-32.png");
    nodeB.setName("华为核心路由器");
    nodeB.setLocation(500, 300);
    box.addElement(nodeB);
    Link link = new Link("link", nodeA, nodeB);
    link.setName("冗余备份线路");
    box.addElement(link);
    link = new Link(nodeA, crm);
    box.addElement(link);
    link = new Link(nodeB, erp);
    box.addElement(link);
    Node cnc = new Node();
    cnc.setLocation(100, 100);
    box.addElement(cnc);
    Node tel = new Node();
    tel.setLocation(200, 100);
    box.addElement(tel);
    SubNetwork chinaTel = new SubNetwork();
    chinaTel.setName("中国电信");
    chinaTel.addChild(tel);
    chinaTel.setLocation(250, 100);
    box.addElement(chinaTel);
    SubNetwork chinaCnc = new SubNetwork();
    chinaCnc.setName("中国网�通");
    chinaCnc.addChild(cnc);
    chinaCnc.setLocation(450, 100);
    box.addElement(chinaCnc);
    link = new Link(nodeA, cnc);
    box.addElement(link);
    link = new Link(nodeB, tel);
    box.addElement(link);
    link = new Link(nodeA, tel);
    box.addElement(link);
    link = new Link(nodeB, cnc);
    box.addElement(link);
    box.setAlarmPropagator(new SummingAlarmPropagator());
    Node node = (Node) box.getElementByID("A");
    AlarmState alarmState = node.getAlarmState();
    alarmState.increaseNewAlarm(AlarmSeverity.CRITICAL, 20);
    node = (Node) box.getElementByID("B");
    alarmState = node.getAlarmState();
    alarmState.increaseNewAlarm(AlarmSeverity.MAJOR, 10);
}
