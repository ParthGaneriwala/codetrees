@Test
public void testHibernate() {
    Session session = sessionFactory.getCurrentSession();
    Transaction tx = session.beginTransaction();
    Consorcio c = new Consorcio("Consorcio", "Echeverria 5640");
    session.save(c);
    Miembro m = new Miembro("Pepe", "pass", c);
    session.persist(m);
    Tag tag = new Tag("tag1");
    session.persist(tag);
    List<Tag> tags = new LinkedList<Tag>();
    tags.add(tag);
    Tarea t = new Tarea("Tarea 1", tags, m);
    session.save(t);
    session.save(m);
    Voto v = new Voto(t, m);
    session.save(v);
    Miembro m2 = new Miembro("Jose", "ppp", c);
    session.save(m2);
    Voto v2 = new Voto(t, m2);
    session.save(v2);
    Tag tag2 = new Tag("tag2");
    session.persist(tag2);
    List<Tag> tags2 = new LinkedList<Tag>();
    tags2.add(tag2);
    Tarea t2 = new Tarea("Tarea 2", tags2, m);
    session.save(t2);
    session.save(m2);
    Proveedor p = new Proveedor("Proveedor 1", "pass");
    session.save(p);
    Oferta o = p.ofertar(t);
    session.save(o);
    session.save(p);
    session.save(t);
    p.addTareaToWatchList(t2);
    session.save(t2);
    session.save(p);
    System.out.println(session.get(Consorcio.class, new Long(1)));
    CalificacionConsorcio cc = p.calificarConsorcio(c, t, new Integer(5));
    session.save(cc);
    CalificacionProveedor cp;
    cp = m2.calificarProveedor(p, t2, new Integer(3));
    session.save(cp);
    CalificacionProveedor cp2;
    cp2 = m.calificarProveedor(p, t2, new Integer(2));
    session.save(cp2);
    System.out.println(session.get(Proveedor.class, new Long(1)));
    tx.commit();
}
