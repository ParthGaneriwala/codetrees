public void iniciaCriaClienteView() {
    tela = new JFrame();
    ouvinte = new acao();
    nome = new JTextField();
    login = new JTextField();
    senha = new JPasswordField();
    senhaConfirmacao = new JPasswordField();
    email = new JTextField();
    cpfCnpj = new JTextField();
    criar = new JButton("Criar");
    criar.addActionListener(ouvinte);
    cancelar = new JButton("Cancelar");
    cancelar.addActionListener(ouvinte);
    loginLabel = new JLabel("Login");
    nomeLabel = new JLabel("Nome");
    cpfCnpjLabel = new JLabel("Cpf/Cnpj");
    senhaLabel = new JLabel("Senha");
    senhaConfirmacaoLabel = new JLabel("Confirmação da Senha");
    emailLabel = new JLabel("Email");
    pessoaFisica = new JRadioButton("Pessoa Fisica");
    pessoaJuridica = new JRadioButton("Pessoa Juridica");
    grupo = new ButtonGroup();
    grupo.add(pessoaFisica);
    grupo.add(pessoaJuridica);
    pessoaFisica.setSelected(true);
    tela.setBounds(500, 300, 400, 300);
    tela.setResizable(false);
    tela.setDefaultCloseOperation(JFrame.DO_NOTHING_ON_CLOSE);
    tela.setTitle("Cria Cliente");
    painel = new JPanel();
    GroupLayout layout = new GroupLayout(painel);
    painel.setLayout(layout);
    layout.setAutoCreateGaps(true);
    layout.setAutoCreateContainerGaps(true);
    organizaFields(layout);
    tela.setLayout(new BorderLayout());
    tela.add(painel, BorderLayout.CENTER);
    tela.setVisible(true);
}
