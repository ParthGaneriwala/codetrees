private void composeDialog() {
    mSelectionList = new JList();
    mSelectionPane = new JScrollPane(mSelectionList);
    JPanel parentPanel = new JPanel();
    mSendBtn = new JButton("Send to this person");
    mSendTeacherBtn = new JButton("Send to teacher");
    mSendGroupBtn = new JButton("Send to group");
    parentPanel.setBorder(BorderFactory.createTitledBorder(BorderFactory.createLineBorder(Color.BLACK), "Recipient List"));
    parentPanel.setLayout(new GridBagLayout());
    mSelectionPane.setPreferredSize(LIST_DIMS);
    GridBagConstraints cns = new GridBagConstraints();
    cns.anchor = GridBagConstraints.NORTH;
    cns.fill = GridBagConstraints.BOTH;
    cns.weightx = cns.weighty = 1.0;
    cns.gridx = cns.gridy = 0;
    cns.insets = new Insets(5, 5, 5, 5);
    parentPanel.add(mSelectionPane, cns);
    cns.gridy++;
    cns.insets = new Insets(0, 5, 0, 5);
    cns.fill = GridBagConstraints.HORIZONTAL;
    cns.weighty = 0;
    parentPanel.add(mSendBtn, cns);
    cns.gridy++;
    parentPanel.add(mSendTeacherBtn, cns);
    cns.gridy++;
    parentPanel.add(mSendGroupBtn, cns);
    add(parentPanel);
    mSendBtn.setEnabled(false);
    AuthenticationHandler handler = AuthenticationHandler.getInstance();
    handler.addObserver(this);
}
