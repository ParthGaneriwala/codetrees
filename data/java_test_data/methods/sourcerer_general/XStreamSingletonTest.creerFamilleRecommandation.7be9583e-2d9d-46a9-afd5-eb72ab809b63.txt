private FamilleRecommandation creerFamilleRecommandation(long i) {
    FamilleRecommandation f = new FamilleRecommandation();
    f.setActive(true);
    f.setId(i * 10 + i);
    f.setImage("/recommandation/image1.jpg");
    f.setLibelle("famille" + (i * 10 + 1));
    f.setOrdreAffichage((int) 1);
    Recommandation reco = new Recommandation();
    reco.setActive(true);
    reco.setDescription("La recommandation 0 de la famille " + i);
    reco.setLibelle("la reco0 de famille " + i);
    reco.setId((long) 1);
    reco.setIdFamilleRecommandation(i);
    reco.setImage("/recommandation/image1.jpg");
    reco.setOrdreAffichage(1);
    addAllConditionMaisonIndividuelle(reco);
    Recommandation reco1 = new Recommandation();
    reco1.setActive(true);
    reco1.setDescription("La recommandation 1 de la famille " + i);
    reco1.setLibelle("la reco1 de famille " + i);
    reco1.setId((long) 1);
    reco1.setIdFamilleRecommandation(i);
    reco1.setImage("/recommandation/image1.jpg");
    reco1.setOrdreAffichage(2);
    addAllConditionMaisonIndividuelle(reco1);
    Recommandation reco2 = new Recommandation();
    reco2.setActive(true);
    reco2.setDescription("La recommandation 2 de la famille " + i);
    reco2.setLibelle("la reco2 de famille " + i);
    reco2.setId((long) 1);
    reco2.setIdFamilleRecommandation(i);
    reco2.setImage("/recommandation/image1.jpg");
    reco2.setOrdreAffichage(3);
    addAllConditionLogementCollectif(reco2);
    f.addRecommandation(reco);
    f.addRecommandation(reco1);
    f.addRecommandation(reco2);
    return f;
}
