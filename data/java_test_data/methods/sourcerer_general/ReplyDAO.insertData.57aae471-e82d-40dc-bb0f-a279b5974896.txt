public String insertData(BoardForm data) throws SQLException {
    Connection conn = null;
    PreparedStatement pstmt = null;
    conn = DbPool.getConnection();
    pstmt = (PreparedStatement) conn.prepareStatement(QUERY_UPDATE_STEP);
    pstmt.setInt(1, data.getRef());
    pstmt.setInt(2, data.getStep());
    pstmt.setInt(3, data.getBoardid());
    logger.debug(pstmt.toString());
    pstmt.executeUpdate();
    pstmt = (PreparedStatement) conn.prepareStatement(QUERY_UPDATE_PSEQ);
    pstmt.setInt(1, data.getPseq());
    pstmt.setInt(2, data.getBoardid());
    logger.debug(pstmt.toString());
    pstmt.executeUpdate();
    data.setStep(data.getStep() + 1);
    data.setLev(data.getLev() + 1);
    data.setPseq(data.getPseq() + 1);
    pstmt = (PreparedStatement) conn.prepareStatement(QUERY_INSERT);
    pstmt.setString(1, data.getWriter());
    pstmt.setString(2, data.getTitle());
    pstmt.setString(3, data.getContent());
    pstmt.setString(4, data.getPassword());
    pstmt.setInt(5, data.getBoardid());
    pstmt.setInt(6, data.getRef());
    pstmt.setInt(7, data.getStep());
    pstmt.setInt(8, data.getLev());
    pstmt.setInt(9, data.getHit());
    pstmt.setInt(10, data.getPseq());
    pstmt.setInt(11, data.getReply());
    logger.debug(pstmt.toString());
    pstmt.executeUpdate();
    pstmt.close();
    conn.close();
    pstmt = null;
    conn = null;

}
