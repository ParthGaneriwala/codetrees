public void testIntersectionPolyTwoSidesOneVertex() {
    Poly p1 = new PolyDefault();
    p1.add(1.0, 3.0);
    p1.add(5.0, 3.0);
    p1.add(5.0, 8.0);
    p1.add(3.0, 6.0);
    p1.add(1.0, 8.0);
    Poly p2 = new PolyDefault();
    p2.add(0.0, 2.0);
    p2.add(2.0, 0.0);
    p2.add(3.0, 1.0);
    p2.add(4.0, 0.0);
    p2.add(6.0, 2.0);
    p2.add(4.0, 4.0);
    p2.add(3.0, 3.0);
    p2.add(2.0, 4.0);
    Poly iexp1 = new PolyDefault();
    iexp1.add(4.0, 4.0);
    iexp1.add(3.0, 3.0);
    iexp1.add(5.0, 3.0);
    Poly iexp2 = new PolyDefault();
    iexp2.add(1.0, 3.0);
    iexp2.add(3.0, 3.0);
    iexp2.add(2.0, 4.0);
    Poly exp = new PolyDefault();
    exp.add(iexp1);
    exp.add(iexp2);
    PolyDefault result = (PolyDefault) Clip.intersection(p1, p2);
    // result.print();
    assertEquals(exp, result);
}
