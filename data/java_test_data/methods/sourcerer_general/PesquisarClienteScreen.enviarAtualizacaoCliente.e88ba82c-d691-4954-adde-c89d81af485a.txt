        // generator.addValue((cliente.getUfRg() == null ? "" : cliente.getUfRg()));
        generator.addValue("&cpf=");
        generator.addValue((Formata.insereMascaraCpfCnpj(cliente.getCpfCnpj()) == null ? "" : (Formata.insereMascaraCpfCnpj(cliente.getCpfCnpj()))));
    }
    generator.addValue("&logradouro=");
    generator.addValue((cliente.getLogradouro() == null ? "" : cliente.getLogradouro()));
    generator.addValue("&endnumero=");
    generator.addValue((cliente.getNumero() == null ? "" : cliente.getNumero()));

    generator.addValue((cliente.getComplemento() == null ? "" : cliente.getComplemento()));
