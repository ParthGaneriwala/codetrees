public void iniciaVisualizarGerenteView(int id, VerGerentesView verGerentesView) throws NumberFormatException, IOException {
    this.verGerentesView = verGerentesView;
    ouvinte = new acao();
    tela = new JFrame();
    tela.setBounds(500, 300, 300, 150);
    tela.setResizable(false);
    tela.setDefaultCloseOperation(JFrame.DO_NOTHING_ON_CLOSE);
    tela.setTitle("Visualizar Gerente");
    painel = new JPanel();
    GroupLayout layout = new GroupLayout(painel);
    painel.setLayout(layout);
    layout.setAutoCreateGaps(true);
    layout.setAutoCreateContainerGaps(true);
    voltar = new JButton("Voltar");
    voltar.addActionListener(ouvinte);
    alterar = new JButton("Alterar");
    alterar.addActionListener(ouvinte);
    gerente = GerenteControlador.get(id);
    nome = new JLabel("Nome do Gerente: ");
    nome1 = new JLabel(gerente.getNome());
    login = new JLabel("Login do Gerente: ");
    login1 = new JLabel(gerente.getLogin());
    int estilo = 0;
    estilo = estilo + Font.ITALIC;
    Font f = new Font(login.getFont().toString(), estilo, 12);
    nome1.setFont(f);
    login1.setFont(f);
    organizaFields(layout);
    tela.setLayout(new BorderLayout());
    tela.add(painel, BorderLayout.CENTER);
    tela.setVisible(true);
}
