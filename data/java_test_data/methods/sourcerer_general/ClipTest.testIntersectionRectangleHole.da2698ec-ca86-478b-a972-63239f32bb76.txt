public void testIntersectionRectangleHole() {
    Poly p1 = new PolyDefault();
    p1.add(0.0, 1.0);
    p1.add(4.0, 1.0);
    p1.add(4.0, 5.0);
    p1.add(0.0, 5.0);
    Poly p2 = new PolyDefault(true);
    p2.add(1.0, 2.0);
    p2.add(3.0, 2.0);
    p2.add(3.0, 4.0);
    p2.add(1.0, 4.0);
    Poly p12 = new PolyDefault();
    p12.add(p1);
    p12.add(p2);
    Poly p3 = new PolyDefault();
    p3.add(2.0, 0.0);
    p3.add(6.0, 0.0);
    p3.add(6.0, 6.0);
    p3.add(2.0, 6.0);
    Poly exp = new PolyDefault();
    exp.add(2.0, 1.0);
    exp.add(4.0, 1.0);
    exp.add(4.0, 5.0);
    exp.add(2.0, 5.0);
    exp.add(2.0, 4.0);
    exp.add(3.0, 4.0);
    exp.add(3.0, 2.0);
    exp.add(2.0, 2.0);
    PolyDefault result = (PolyDefault) Clip.intersection(p12, p3);
    // result.print();
    assertEquals(exp, result);
}
