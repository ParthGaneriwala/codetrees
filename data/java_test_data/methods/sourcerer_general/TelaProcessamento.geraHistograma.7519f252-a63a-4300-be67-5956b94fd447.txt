private void geraHistograma() {
    Histograma hist = null;
    hist = new Histograma(this.imagemA);
    hist.desenhaGrafico();
    hist.desenhaEqualizado();
    BufferedImage graf = hist.getGrafico();
    JFrame frame = new JFrame();
    frame.setLayout(null);
    ImageIcon icon = new ImageIcon(graf);
    JLabel label = new JLabel(icon);
    frame.add(label);
    label.setBounds(10, 10, graf.getWidth(), graf.getHeight());
    BufferedImage equa = hist.getGraficoEqualizado();
    ImageIcon icon2 = new ImageIcon(equa);
    JLabel label2 = new JLabel(icon2);
    frame.add(label2);
    label2.setBounds(320, 10, graf.getWidth(), graf.getHeight());
    this.imagemResult = hist.getImagemEqualizada();
    this.atualizaResultado();
    JLabel textoImagemA = new JLabel("Histograma da imagem normal");
    frame.add(textoImagemA);
    textoImagemA.setBounds(20, 420, 200, 20);
    JLabel textoImagemB = new JLabel("Histograma da imagem equalizada");
    frame.add(textoImagemB);
    textoImagemB.setBounds(330, 420, 200, 20);
    int largura = graf.getWidth() + equa.getWidth() + 40;
    int altura = graf.getHeight() + 80;
    Dimension d = new Dimension(largura, altura);
    frame.setSize(d);
    frame.setPreferredSize(d);
    frame.setResizable(false);
    frame.pack();
    frame.setVisible(true);
}
