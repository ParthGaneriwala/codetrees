private Track createSNPTrack(int inXAxisStartValue, int inXAxisEndValue) {
    Track track = new Track("Mouse SNPs");
    track.setXAxisStartValue(inXAxisStartValue);
    track.setXAxisEndValue(inXAxisEndValue);
    track.setShowId(false);
    Gene2D gene = new Gene2D();
    gene.addExon(new Exon(130, 130, Strand.FORWARD));
    gene.setId("Mouse SNP 130");
    track.addGene(gene);
    gene = new Gene2D();
    gene.addExon(new Exon(134, 134, Strand.FORWARD));
    gene.setId("Mouse SNP 134");
    track.addGene(gene);
    gene = new Gene2D();
    gene.addExon(new Exon(234, 234, Strand.FORWARD));
    gene.setId("Mouse SNP 234");
    track.addGene(gene);
    gene = new Gene2D();
    gene.addExon(new Exon(260, 260, Strand.FORWARD));
    gene.setId("Mouse SNP 260");
    track.addGene(gene);
    gene = new Gene2D();
    gene.addExon(new Exon(700, 700, Strand.REVERSE));
    gene.setId("Mouse SNP 700");
    track.addGene(gene);
    gene = new Gene2D();
    gene.addExon(new Exon(720, 720, Strand.REVERSE));
    gene.setId("Mouse SNP 720");
    track.addGene(gene);
    return track;
}
