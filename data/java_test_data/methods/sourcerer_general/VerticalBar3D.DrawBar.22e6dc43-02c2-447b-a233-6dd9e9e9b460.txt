public void DrawBar(Graphics backg, int x, int y, int w, int h, Color color) {
    double w1 = -w * 0.1;
    double x1 = x + 2.0 * w1;
    double w2 = w * 0.6;
    int nr_puncte = 4;
    int[] v1_x = new int[nr_puncte];
    int[] v1_y = new int[nr_puncte];
    int[] v2_x = new int[nr_puncte];
    int[] v2_y = new int[nr_puncte];
    v1_x[0] = (int) (x1);
    v1_x[1] = (int) (x1 + w2);
    v1_x[2] = (int) (x1 + 4 * w2 / 3);
    v1_x[3] = (int) (x1 + w2 / 3);
    v1_y[0] = (int) y;
    v1_y[1] = (int) y;
    v1_y[2] = (int) (y - w2 / 3);
    v1_y[3] = (int) (y - w2 / 3);
    v2_x[0] = (int) (x1 + w2);
    v2_x[1] = (int) (x1 + w2);
    v2_x[2] = (int) (x1 + 4 * w2 / 3);
    v2_x[3] = (int) (x1 + 4 * w2 / 3);
    v2_y[0] = (int) y;
    v2_y[1] = (int) y + h;
    v2_y[2] = (int) (y + h - w2 / 3);
    v2_y[3] = (int) (y - w2 / 3);
    g2b.setColor(color);
    g2b.setComposite(AlphaComposite.getInstance(AlphaComposite.SRC_OVER, transparency));
    g2b.fillRect((int) x1, y, (int) w2, h);
    g2b.setColor(color.brighter());
    g2b.fillPolygon(v1_x, v1_y, nr_puncte);
    g2b.setColor(color.darker());
    g2b.fillPolygon(v2_x, v2_y, nr_puncte);
    g2b.setColor(Color.black);
    g2b.drawRect((int) x1, y, (int) w2, h);
    g2b.drawPolygon(v1_x, v1_y, nr_puncte);
    g2b.drawPolygon(v2_x, v2_y, nr_puncte);
    g2b.setComposite(AlphaComposite.getInstance(AlphaComposite.SRC_OVER, 1.f));
}
