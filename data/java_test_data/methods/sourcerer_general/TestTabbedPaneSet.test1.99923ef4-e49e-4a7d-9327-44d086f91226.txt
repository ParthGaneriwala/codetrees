// Some browsers don't like a textarea written as an empty tag. Hence we have
@Test
public void test1() throws Exception {
    HTMLDoc doc = new HTMLDoc();
    HTML html = new HTML("TabbedPaneSet Examples");
    doc.setRootNode(html);
    html.getHead().addStyleTag(TabbedPaneSet.generateCSS());
    html.getHead().addJavascript(TabbedPaneSet.generateJavascript());
    TabbedPaneSet tabSet1 = new TabbedPaneSet();
    Span infoTab = new Span();
    infoTab.addImage("person.gif");
    infoTab.addContent(" User Info");
    tabSet1.addTab(infoTab, generateFormA());
    Span contactTab = new Span();
    contactTab.addImage("contact.gif");
    contactTab.addContent("Contact Info");
    tabSet1.addTab(contactTab, generateFormB(), true);
    Span settingsTab = new Span();
    settingsTab.addImage("settings.gif");
    settingsTab.addContent(" Settings");
    Div div = new Div();
    div.br(2);
    div.addContent("Comments:");
    div.br();
    div.addTextarea("foo", "").setCols(80).setRows(8);
    tabSet1.addTab(settingsTab, div);
    TabbedPaneSet tabSet2 = new TabbedPaneSet().useRoundedTabs(false);
    tabSet2.addTab("Tab 1", new InputButton("Button 1"));
    tabSet2.addTab("Tab 2", new InputText("foo", "bar").setSize(30));
    div = new Div("Pane 3").setId("foo").setStyle(CSS.bgColor(HTMLColor.ANTIQUE_WHITE));
    div.br(10);
    div.addCenter("---------------------------------------------------------------");
    tabSet2.addTab("Tab 3", div);
    tabSet2.addTab("InnerTabset", generatePane4());
    TabbedPaneSet tabSet3 = new TabbedPaneSet(Orientation.VERTICAL);
    tabSet3.addTab("A", new InputButton("Button 1"));
    tabSet3.addTab("B", new InputText("foo", "bar").setSize(30));
    div = new Div("Pane 3").setId("foo").setStyle(CSS.bgColor(HTMLColor.ANTIQUE_WHITE));
    div.br(10);
    div.addCenter("---------------------------------------------------------------");
    tabSet3.addTab("C", div, false, "alert('Congratulations on selecting C!')");
    Body body = html.getBody().setBackgroundColor(HTMLColor.BEIGE);
    body.setOnLoad(tabSet1.generateOnLoad() + tabSet2.generateOnLoad() + tabSet3.generateOnLoad());
    body.br();
    body.addSubtag(tabSet1);
    body.br();
    body.addSpan("The Java to generate this tab set looks like:");
    body.addDiv().setStyle("background-color: #cccccc; border: 1px solid black; width: 600px;").addPre("      html.getHead().addStyle(TabbedPaneSet.generateCSS());\n" + "      html.getHead().addJavascript(TabbedPaneSet.generateJavascript());\n" + "\n" + "      TabbedPaneSet tabSet1 = new TabbedPaneSet();\n" + "\n" + "      Span infoTab = new Span();\n" + "      infoTab.addImage(\"person.gif\");\n" + "      infoTab.addContent(\" User Info\");\n" + "      tabSet1.addTab(infoTab, generateFormA());\n" + "\n" + "      Span contactTab = new Span();\n" + "      contactTab.addImage(\"contact.gif\");\n" + "      contactTab.addContent(\"Contact Info\");\n" + "      tabSet1.addTab(contactTab, generateFormB(), true);\n" + "\n" + "      Span settingsTab = new Span();\n" + "      settingsTab.addImage(\"settings.gif\");\n" + "      settingsTab.addContent(\" Settings\");\n" + "      Div div = new Div();\n" + "      div.br(2);\n" + "      div.addContent(\"Comments:\");\n" + "      div.br();\n" + "      div.addTextarea(\"foo\", \"\").setCols(80).setRows(8);\n" + "      tabSet1.addTab(settingsTab, div);\n" + "\n" + "      body.addSubtag(tabSet1);\n" + "      body.setOnLoad(tabSet1.generateOnLoad());\n");
    body.br(2);
    body.addSpan("Ok. How about two tabbed pane sets that can coexist on the same page?");
    body.br();
    body.addSpan("Not good enough? How about a tabbed pane set inside another tabbed pane set? With checkboxes!");
    body.br(2);
    body.addSubtag(tabSet2);
    body.br(2);
    body.addSpan("Here is a vertical tabbed pane set. Tab 'C' has a 'onclick' defined.");
    body.br(2);
    body.addSubtag(tabSet3);
    File testFile = new File("src/com/hfg/html/custom/doc-files/TabbedPaneTest.html");
    PrintWriter writer = new PrintWriter(new FileWriter(testFile));
    doc.toIndentedHTML(writer, 1, 3);
    writer.close();
}
